    嵌套

    /*1.直接嵌套*/ 
.box {
    font-size:25px;
    .per  {
    color:#f60;
    }
    &>.abc {
    background-color: #666;
    }
}
/*1结果*/
.box {
    font-size:25px;
}
.box .per  {
    color:#f60;
}
.box>.abc  {
    background-color: #666;
}
// -----------------------------分割线----------------------
/* 2. 使用font@at-root */
.box {
  font-size:25px;
  @at-root{
    .per {
      color:#f60;
    }
    &>.abc {
      background-color: #666;
    }
  }
}
// 2结果
.box {
  font-size:25px;
}
.per {
  color:#f60;
}
.box>.abc {
  background-color: #666;
}

    变量

    关键字(以$开头)： $ 变量名： 自定义 赋值符号： : 变量值： css样式值

    // 声明变量
$myFontSize: 50px;

// 使用
div {
    width: $myFontSize;
}

// 编译结果
div  {
    width: 50px;
}

    混合宏@mixin

    关键字： @mixin 名称： 自定义

    // 1.不带参数
// 声明一个混合宏@mixin
@mixin myBorderRadius{
    -webkit-border-radius:5px;
    border-radius:5px;
}
// 使用
.box  {
    @include myBorderRadisu;
}
// 1编译结果
.box {
    -webkit-border-radius:5px;
    border-radius:5px
}
// 2.
// 声明一个带有参数的宏：
@mixin 宏名称($abc,$w,$h){    // ————此处可以设置默认值$abc:5px;
  -webkit-border-radius:$abc;
  border-radius:$abc;
  margin-top: -($h) / 2;  // ———— 可以使用运算
  margin-left: -($w) / 2; // ———— 可以使用运算
}
// 如果参数过多可以使用：  ...
@mixin box-shadow($shadows...){
  @if length($shadows) >= 1 {     // —— 如果参数的length大于1
    -webkit-box-shadow: $shadows;
    box-shadow: $shadows;
  } @else {
    $shadow: 0 0 2px rgba(#000,.25);
    -webkit-box-shadow: $shadow;
    box-shadow: $shadow;
  }
}

    @extend继承

    // 例如
.box {
  color:#f60;
}
// .per类名继承.box类名
.per {
  font-size:16px;
  @extend .box;
}
// 编译结果
.box,.per {
  color:#f60;
}
.per {
  font-size:16px;
}

    %占位符

    关键字： %

    // 声明一个占位符空间
%mt5 {
  margin-top: 5px;
}
// 用@extend使用占位符
.btn  {
  font-size:25px;
  @extend %mt5;
}
.per  {
  @extend %mt5;
}
// 编译结果
.btn,.per  {
  margin-top: 5px;
}
.btn  {
  font-size:25px;
}

    在vue中使用SCSS

    在vue中安装： npm install node-sass sass-loader –save-dev
    npm安装sass时的注意：

    1 __ 如果安装过慢。
     切换npm源，可大幅度提升速度
     npm install -g mirror-config-china --registry=http://registry.npm.taobao.org
2 __如果安装完毕依赖后。项目编译错误。
     则将package.json中将sass-loader的版本改为7.3.1，官方解释为版本过高导致的编译错误，并重新npm i
     "sass-loader": "^7.3.1",

    使用

    // 将style标签上添加属性 lang="scss"
<style lang="scss" scoped>
    @import "../../../../static/css/common.scss";

    div  {
        width: 120px;
        .abc {
            font-size: 16px;
        }
    }
</style>

    在webstorm中使用SCSS

    在webstorm中配置自动编译：
    英文版：依次点开 file-->>Settings-->>tool-->>File--Watchers-->>右上角加号-->>scss
    中文版：依次点开 文件-->>设置-->>工具-->>File--Watchers-->>右上角加号-->>scss

    配置：
    更改Programs输入框内容：
        改成ruby地址中的bin目录内的scss.bat的路径，如果安装ruby
        时没有更改安装路径则默认地址如下：
            C:\Ruby26-x64\bin\sass.bat    复制本条地址替换
            -->>点击应用-->>点击确认
    《完成》
    下次使用webstorm编写scss文件时。将自动在同级目录下实时生成同名的.css文件
