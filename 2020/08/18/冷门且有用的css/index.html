    多选按钮自定义背景色

    input[type=checkbox]{
  box-sizing: border-box;
  padding: 0;
  margin: 0;
  cursor: pointer;
  position: relative;
  width: 20px;
  height: 20px;
  font-size: 12px;
  overflow: visible;
  &:after{
    box-sizing: border-box;
    padding: 0;
    margin: 0;
    width: calc(100% + 2px);
    height: calc(100% + 2px);
    color: transparent;
    position: absolute;
    top: calc(50% - 1px);
    left: calc(50% - 1px);
    font-size: 12px;
    line-height: 20px;
    font-weight: bold;
    transform: translateX(calc(-50% + 1px)) translateY(calc(-50% + 1px));
    background-color: #1F253D;
    text-align: center;
    content: " ";
    border-radius: 4px;
  }
  &:checked::after{
    content: "✓";
    color: #fff;
  }
}

    更改“自动联想的input”样式

    input[type=text], input[type=password], input[type=number]  {
  /* 字体色 */
  &:-internal-autofill-previewed, &:-internal-autofill-selected  {
    -webkit-text-fill-color: #fff;
  }
  /* 背景色 */
  &:-webkit-autofill  {
    box-shadow: 0 0 0px 1000px #2d3a4b inset !important;
  }
}

    隐藏，input[type=number]的右面角标

    input[type='number']{
  -moz-appearance:textfield;
  &::-webkit-inner-spin-button,
  &::-webkit-outer-spin-button {
    -webkit-appearance: none;
    margin: 0;
  }
}

    设置滚动条样式

    /* 示例： */
    /* 滚动条样式 */
    .abc::-webkit-scrollbar {
        width: 4px;
    }
    /* 滑块样式 */
    .abc::-webkit-scrollbar-thumb {
        border-radius: 10px;
        -webkit-box-shadow: inset 0 0 5px rgba(0,0,0,0.2);
        background: rgba(0,0,0,0.2);
    }
    /* 滑块的轨道样式 */
    .abc::-webkit-scrollbar-track {
        -webkit-box-shadow: inset 0 0 5px rgba(0,0,0,0.2);
        border-radius: 5px;
        background: rgba(0,0,0,0.1);
    }
/* ------------------------------------------------------- */
/* 具体含义及其他设置项： */
    ::-webkit-scrollbar {}            /* 滚动条整体部分 */
    ::-webkit-scrollbar-thumb {}    /* 滚动条里面的滑块 */
    ::-webkit-scrollbar-track {}    /* 滑块的轨道 */
    ::-webkit-scrollbar-button {}    /* 滚动条的轨道的两端按钮 */
/* ------------------------------------------------------- */
/* 他们的关系是: */
    // scrollbar包含三个直接子元素：
    //     start-button
    //     track
    //     end-button
    // track包含一个直接子元素：
    //     thumb
/* ------------------------------------------------------- */
/* 使用伪类： */
    // scrollbar、track、thumb、button都可以使用:hover和active等伪类
/* 例如： */
    .abc::-webkit-scrollbar-thumb:hover  {
        background-color: #fff;
    }
/*-------------------------------------------------------*/
/* 常用设置：*/
    /* 滚动条样式, 主要作用 可以设置条的宽度(垂直条的宽度)，高度(水平条高度) */
    .abc::-webkit-scrollbar {
        width: 4px;
        height: 4px;
    }
    /* 只设置垂直滚动条样式，可以轨道的宽度，背景色 */
    .abc::-webkit-scrollbar:vertical  {}
    /* 只设置水平滚动条样式，可以轨道的高度，背景色 */
    .abc::-webkit-scrollbar:horizontal  {}


    /* 滑块样式 */
    .abc::-webkit-scrollbar-thumb {}


    /* 滑块的轨道条样式，只可以控制背景色圆角之类样式，不可以控制宽高 */
    .abc::-webkit-scrollbar-track {}


    /* 设置条两边按钮样式 */
    .abc::-webkit-scrollbar-button  {}
    /* 只设置上/左按钮样式 */
    .abc::-webkit-scrollbar-button:start  {}
    /* 只设置下/右按钮样式 */
    .abc::-webkit-scrollbar-button:end  {}
/* ------------------------------------------------------- */
