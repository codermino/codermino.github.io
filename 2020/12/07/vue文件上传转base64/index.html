    <template>
  <div class="imageUpload">

      :auto-upload="false"

</template>

<script>
import acceptFileType from '../util/acceptFileType'
export default {
  name: "imageUpload",
  data() {
    return {
      acceptTypeList: []
    }
  },
  methods: {
    acceptFile(type){
      this.acceptTypeList = acceptFileType(type)
      return this.acceptTypeList
    },
    changeFile(file, fileList) {
      console.log(file.raw)
      const fileName = file.name;
      const m = fileName.match(/\.(\w+)(#|\?|$)/);
      const fileType = (m && m[1]).toString().toLowerCase();
      if (!this.acceptTypeList.includes(fileType)){
        this.$message.info('上传的文件格式不对!');
        return;
      }
      const isLt5M = file.size / 1024 / 1024 < 5;
      if (!isLt5M) {
        this.$message.info('上传头像图片大小不能超过 5MB!');
        return;
      }
      let This = this;
      let reader = new FileReader();
      reader.readAsDataURL(file.raw);
      //开始转
      reader.onload = function(e){
        this.result; //base64编码
        console.log(this.result)
        console.log(e.target.result)
      }
      //转 失败
      reader.onerror = function (error) {
        console.log('Error: ', error)
      }
      //转 结束  咱就 resolve 出去
      reader.onloadend = function() {
        console.log('转换结束')
      };
    }
  }
}
</script>

    acceptFileType.js文件

    export default function acceptFile(e) {
  const allowHook = {
    video: '.mp4, .ogv, .ogg, .webm',
    audio: '.wav, .mp3, .ogg, .acc, .webm, .amr',
    file: 'doc,.docx,.xlsx,.xls,.pdf',
    excel: 'xlsx,.xls',
    img: '.jpg, .jpeg, .png, .gif'
  }
  if (e) {
    return allowHook[e];
  }
  let srt = null
  for (const k in allowHook) {
    srt += allowHook[k]
  }
  return srt
}

    文件类型验证 Demo2

    /*
*  校验上传文件
* */
export default function validUploadFile(type = 'img', file) {
  if (!file) {
    file = type;
    type = 'img'
  }
  const fileName = file.name;
  const m = fileName.match(/\.(\w+)(#|\?|$)/);
  const fileType = (m && m[1]).toString().toLowerCase();
  const allowHook = {
    video: ['mp4', 'ogv', 'ogg', 'webm'],
    audio: ['wav', 'mp3', 'ogg', 'acc', 'webm', 'amr'],
    file: ['doc', 'docx', 'xlsx', 'xls', 'pdf'],
    excel: ['xlsx', 'xls'],
    img: ['jpg', 'jpeg', 'png', 'gif']
  }
  const validType = (allowHook[type] || []).includes(fileType);
  if (!validType) {
    const supprtTypes = allowHook[type].join(',');
    return `只支持${supprtTypes}类型文件上传`;
  }
  if (fileName.indexOf('%') > -1 || fileName.indexOf('&') > -1) {
    return '上传文件名称不能带有字符"%","&"';
  }
  const isLt10M = file.size / 1024 / 1024 < 10;
  if (type === file && !isLt10M) {
    this.$message.error('上传资料大小不能超过 10MB!');
    return false;
  }
  return true;
}
