    使用原生的input

    <template>
  <div class="imageUpload">
    <input id="fileId1" type="file" accept="image/png,image/gif,'image/jpeg'" name="file" @change="fileChange"/>
    <img v-if="imageUrl" :src="imageUrl" class="uploadAvatar"/>
  </div>
</template>

<script>
export default {
  name: "imageUpload",
  data() {
    return {
      imageUrl: '',
      imageBaseUrl: '',
    }
  },
  methods: {
    fileChange (e){
      const file = e.target.files[0]
      const isJPGORPNG = file.type === 'image/jpeg' || file.type === 'image/png'
      const isLt1M = file.size / 1024 / 1024 < 1;
      if (!isJPGORPNG) {
        this.$message.info('上传头像图片只能是 JPG 或 PNG 格式!');
        return;
      }
      if (!isLt1M) {
        this.$message.info('上传头像图片大小不能超过 1MB!');
        return;
      }
      let This = this;
      let reader = new FileReader()
      reader.readAsDataURL(file);
      reader.onload = function(e){
        this.result; //base64编码
        console.log(this.result)
        This.imageBaseUrl = this.result;
        This.imageUrl = this.result;
      }
      reader.onerror = function (error) {
        console.log('Error: ', error)
      }
    }
  }
}
</script>

    使用el-upload

    <template>
  <div class="imageUpload">
      <el-upload
            class="avatar-uploader"
            action=""
            ref="uploadAvatar"
            :show-file-list="false"
            :auto-upload="false"
            :on-change="changeFile">
            <img v-if="imageUrl" :src="imageUrl" class="uploadAvatar">
            <i v-else class="el-icon-plus avatar-uploader-icon"></i>
      </el-upload>
  </div>
</template>

<script>
export default {
  name: "imageUpload",
  data() {
    return {
      imageUrl: '',
      imageBaseUrl: '',
    }
  },
  methods: {
    /**
     * 文件框改变事件
     * @param file
     * @param fileList
     */
    changeFile(file, fileList) {
      const isJPGORPNG = (file.raw.type === 'image/jpeg' || file.raw.type === 'image/png');
      const isLt1M = file.size / 1024 / 1024 < 1;

      if (!isJPGORPNG) {
        this.$message.info('上传头像图片只能是 JPG 或 PNG 格式!');
        return;
      }
      if (!isLt1M) {
        this.$message.info('上传头像图片大小不能超过 1MB!');
        return;
      }

      let This = this;
      let reader = new FileReader();
      reader.readAsDataURL(file.raw);
      reader.onload = function(e){
        this.result; //base64编码
        console.log(this.result)
        This.imageBaseUrl = this.result;
        This.imageUrl = this.result;
      }
      reader.onerror = function (error) {
        console.log('Error: ', error)
      }
    }
  }
}
</script>

    tip

    在上传图片时, accept不要设置成 accept="image/png,image/jpg,image/jpeg"!
后果: 浏览图片时不会自动进入相册,默认打开的是最近使用文档文件夹 
但是最近的显示为空,而且图片会变灰选择不了.
直接设置成 accept="image/*" ;
