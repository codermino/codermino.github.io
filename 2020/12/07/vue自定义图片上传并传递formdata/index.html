      beforeRemove(file, fileList,_id) {
  // 在fileList中查询是否包含选择要删除的图片

      confirmButtonText: '确定',
      cancelButtonText: '取消',
      type: 'warning'

      this.delPic(file,fileList,_id) // 在这里真正的处理图片列表
    }).catch(() => {
      this.$message({
        type: 'info',
        message: '已取消删除'
      });
    });
    return false; // 这是重点,不管上面的操作结果如何都返回false
  }
},
async delPic(file, fileList, _id) {
  const file_split = file.url.split('/');
  const name = file_split[file_split.length - 1];
  // 传递给后端的数据，后端进行图片删除
  const result = await this.axios.post('http://localhost:3000/data/delete_pic_f', {_id, name});
  if (result.data.code === 200){
    // 前端的fileList过滤掉删除的图片
    this.fileList = this.fileList.filter((item, i) => {
      return item.url !== file.url;
    });
    // 重新获取列表数据
    this.getList();
    this.$message.success(result.data.data.msg);
  } else{
    this.$message.error(result.data.msg);
  }
},

uploadProductPicBefore(file) {
  const fileType = file.type;
  const is3M = file.size / 1024 / 1024 < 3; // 限制小于3M
  let isJpg = false;
  if (fileType === 'image/jpeg' || fileType === 'image/png' || fileType === 'image/gif' ||  fileType === 'image/bmp') {
    isJpg = true;
  }
  if (!isJpg) {
    this.$message({
      message: '上传的图标只能是jpg、png、gif、bmp格式!',
      type: 'warning'
    });
  }
  if (!is3M) {
    this.$message({
      message: '上传大小不能超过3M!',
      type: 'warning'
    });
  }
  this.isDialog = isJpg && is3M;
  return is3M && isJpg;
}

uploadProductPic: async function (file) {
  const that = this;
  const reader = new FileReader();
  if (file) {
    reader.readAsDataURL(file.file)
  }
  const id = file.data.id;
  reader.onload = function (e) {
    const base64Code=this.result;
    // 进行压缩，传递参数为base64文件和callback函数
    compress(base64Code, async (file) => {
      // 创建一个FormData
      const formData = new FormData();
      // 添加file属性，后台使用file字段进行接收
      formData.append("file", file);
      // 添加data属性，后台使用data字段进行接收
      formData.append('data', id);
      // 向后盾发起请求，并且将headers改为form-data

    接收一个base64

    // 压缩图片
function compress(base64,callback) {
  const img = new Image();
  img.src = base64;
  img.onload = () => {
    const w = img.width;
    const h = img.height;
    const quality = 0.5;  // 默认图片质量为0.92
    // 生成canvas
    const canvas = document.createElement('canvas');
    const ctx = canvas.getContext('2d');
    // 创建属性节点
    const anw = document.createAttribute("width");
    anw.nodeValue = w;
    const anh = document.createAttribute("height");
    anh.nodeValue = h;
    canvas.setAttributeNode(anw);
    canvas.setAttributeNode(anh);

    // 铺底色 PNG转JPEG时透明区域会变黑色
    ctx.fillStyle = "#fff";
    ctx.fillRect(0, 0, w, h);

    ctx.drawImage(img, 0, 0, w, h);
    // quality值越小，所绘制出的图像越模糊
    const new_base64 = canvas.toDataURL('image/jpeg', quality); // 图片格式jpeg或webp可以选0-1质量区间
    callback(convertBase64UrlToBlob(new_base64));
    // callback(new_base64);

    // 返回base64转blob的值
    // console.log(`原图${(base64.length / 1024).toFixed(2)}kb`, `新图${(new_base64.length / 1024).toFixed(2)}kb`);
  }
}
// 将base64的图片转换为file文件
function convertBase64UrlToBlob(urlData) {
  let bytes = window.atob(urlData.split(",")[1]); //去掉url的头，并转换为byte
  //处理异常,将ascii码小于0的转换为大于0
  let ab = new ArrayBuffer(bytes.length);
  let ia = new Uint8Array(ab);
  let type = urlData.split(",")[0].match(/:(.*?);/)[1];
  for (let i = 0; i < bytes.length; i++) {
    ia[i] = bytes.charCodeAt(i);
  }
  return new Blob([ab], { type: type });
}

    FormData的主要用途有两个：
    1、将form表单元素的name与value进行组合，实现表单数据的序列化，从而减少表单元素的拼接，提高工作效率。
    2、异步上传文件
    (创建一个空对象,可以通过append()方法来追加数据)
