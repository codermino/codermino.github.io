    md5.js

    const crypto = require("crypto");
module.exports = function(password){
  let md5 = crypto.createHash('md5');
  return md5.update(password).digest('base64');
};

    使用express-session中间件

    使用express-session中间件,并且如果用户主动刷新页面那么session自动重新计时，并且session的有效时间为60s
app.use(session({
  secret: 'keyboard cat',
  name : 'login',
  resave: true,//是否允许session重新设置
  saveUninitialized: false,
  cookie: {maxAge: 60 * 1000},
  rolling: true   //add 刷新页面 session 过期时间重置
}));

    注册接口

    app.post('/dologin',function (req,res) {
  let form = new formidable.IncomingForm();
  form.parse(req, function(err, fields, files){
    const array = [];
    const username = fields.dengluming;
    let password = fields.mima;
    password = (你的md5加密方式) 例如:md5(password);
    const url = "mongodb://localhost:27017/";
    //连接数据库
    MongoClient.connect(url, {useUnifiedTopology:true}, (err, client)=> {
      //回调函数表示链接成功做的事情，db参数就是连接上的数据库实体
      if(err) {
        console.log("数据库连接失败");
        return ;
      }
      //插入数据，集合如果不存在，也没有关系，程序会帮你创建
      const db = client.db('learnnode');
      const result = db.collection('Student').find({username:username});
      result.each(function (err, doc) {
        if(err) {
          return;
        }
        if(doc != null) {
          array.push(doc); //放入结果数组
        }else {
          let checkNum = 0;
          const resultNum = array.length;
          if(resultNum === 0){
            res.send('-2');
          }else {
            //遍历结束，没有更多的文档
            array.forEach(function (data) {
              if(data.password === password){
                req.session.login = true;
                req.session.username = username;
                req.session.password = password;
                res.send('1');
              }else {
                checkNum++;
              }
            });
            if(checkNum === resultNum){
              res.send('-1');
            }
          }
        }
      });
    });
  });
});

    注册接口

    app.post('/doregist',function (req,res) {
  let form = new formidable.IncomingForm();
  form.parse(req, function(err, fields, files){
    const username = fields.username;
    let password = fields.password;
    const url = "mongodb://localhost:27017/";
    MongoClient.connect(url, {useUnifiedTopology:true}, (err, client)=> {
      //回调函数表示链接成功做的事情，db参数就是连接上的数据库实体
      if(err) {
        console.log("数据库连接失败");
        return ;
      }
      //插入数据，集合如果不存在，也没有关系，程序会帮你创建
      const db = client.db('learnnode');
      password = md5(md5(password).substr(4,7) + md5(password));
      const myObj =
        {username: username, password: password};
      db.collection('Student').insertOne(myObj,function (err,result) {
        if(err){
          console.log(err);
          res.send('-1');
        }else {
          console.log('插入数据成功');
          res.send('1');
        }
        client.close();
      });
    });
  });
});
