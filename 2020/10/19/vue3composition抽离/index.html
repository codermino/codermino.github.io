    composition API的好处

    1. 方便代码抽离
      可以将属于同一个业务功能的属性和方法等抽离出去方便管理

    抽离成一个函数

    const peopleProp = ()=>{
    const age = ref(18)
    const state = reactive({
      count:0
    })

    watch(
      [age,()=>state.count],
      ([newAge, newCount], [oldAge, oldCount]) => {
        console.log(newAge, newCount);
      }
    )

    const changeValue = ()=>{
      age.value = 20
      state.count ++
    }

    return{
      state,
      age,
      changeValue
    }
  }

    引用

    const {state,age,changeValue,changeEmployee} = peopleProp()

    导出

    return{
    ...toRefs(state),
    age,
    changeValue,
    changeEmployee
  }

    抽离成单独的文件
    抽离

    import {ref,reactive,watch} from 'vue'

const peopleProp = ()=>{
  const age = ref(18)
  const state = reactive({
    count:0,
    employee:[
      {name:'张三',sex:'男'},
      {name:'张三2',sex:'男'},
      {name:'张三3',sex:'男'},
      {name:'张三4',sex:'男'},
      {name:'张三5',sex:'男'}
    ]
  })

  watch(
    [age,()=>state.count],
    ([newAge, newCount], [oldAge, oldCount]) => {
      console.log(newAge, newCount);
    }
  )

  const changeValue = ()=>{
    age.value = 20
    state.count ++
  }

  const changeEmployee = ()=>{
    state.employee[2].name = '孙赫'
    console.log(state.employee);
  }

  return{
    state,
    age,
    changeValue,
    changeEmployee
  }
}

export default peopleProp

    引入

    const {state,age,changeValue,changeEmployee} = peopleProp()

    导出

    return{
    ...toRefs(state),
    age,
    changeValue,
    changeEmployee
  }
