    vue图片未加载完成显示占位图
    2020/10/19 vue3 vue图片未加载完成显示占位图 

    <template>
  <div class="container">
    <img v-if="url" :src="url" alt="" @load="imgLoad">
    <img v-else src="../assets/placeholder-image.jpg" alt="">
  </div>
</template>

<script>
import {reactive,toRefs} from 'vue'
export default {
  name: "showImg",
  setup(){
    const state = reactive({
      url:null,
      showImg:false
    })

    setTimeout(()=>{
      state.url = require("../assets/logo.png")
    },3000)

    const imgLoad = ()=>{
      state.showImg = true
    }

    return{
      ...toRefs(state),
      imgLoad
    }
  }
}
</script>

<style scoped>
 .container{
   width: 100px;
   height: 100px;
 }
  .container img{
    width: 100%;
  }
</style>
