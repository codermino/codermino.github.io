    vue的$set使用总结

    <template>
  <div class="setprac">
    <div v-for="(item,index) in userdata" :key="index">
      <span>name:{{item.name}}</span>
      <span> age:{{item.age}}</span>
      <span v-if="item.sex"> sex:{{item.sex}}</span>
      <AButton @click="changeDataName(item.name,index)">更改姓名</AButton>
      <AButton @click="changeDataSex(item,index)">添加性别</AButton>
      <AButton @click="changeDataItem(item,index)">更改该条数据</AButton>
    </div>
    <hr>
    <div>{{user.name}}---{{user.age}}</div>
    <a-button @click="addSex">添加sex</a-button>
    {{user}}
    <hr>
    <span v-for="(item,index) in numList" :key="'num'+index">
      {{item}}
    </span>
    <a-button @click="changeNumList">改变数字数组的值</a-button>
    <a-button @click="changeNumListSingle">改变单个数字数组的值</a-button>
  </div>
</template>

<script>
export default {
  name: "setprac",
  data() {
    return {
      userdata: [
        {name:'zs',age:11},
        {name:'ls',age:12},
        {name:'ww',age:13}
      ],
      user:{name:'sunhe',age:18},
      numList:[1,2,3,4,5]
    }
  },
  methods: {
    changeDataName(name,index) {
      // 修改本来存在的字段不需要使用$set也可以触发更新
      this.userdata[index].name = 'update'
    },
    changeDataSex(item,index){
      // this.userdata[index].sex = '男'
      this.$set(this.userdata[index],'sex','男')
    },
    addSex(){
      // this.user.sex = '男'
      this.$set(this.user,'sex','男')
    },
    changeNumList(){
      this.numList = [6,7,8,9,10]
    },
    changeNumListSingle(){
      // this.numList[0] = 10
      this.$set(this.numList,0,10)
    },
    changeDataItem(item,index){
     // this.userdata[index] = {name:'zl',age:100}
      this.$set(this.userdata,index,{name:'zl',age:100})
    }
  },
}
</script>
