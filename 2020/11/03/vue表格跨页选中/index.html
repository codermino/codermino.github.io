    getData(val){
      this.currentPage = val
      this.$nextTick(()=>{
        this.setChecked()
      })
    },
    // 全选和取消全选时触发
    handleSelectAll (selection) {
      if (selection.length){
        let data = this.$refs.purchaseTable.data
        data.forEach((item) => {
          this.selectedItems.add(item)
        })
      }
      else{
        let data = this.$refs.purchaseTable.data
        data.forEach(item=>{
          for (let item2 of this.selectedItems) {
            if (item.id === item2.id){
              this.selectedItems.delete(item2)
            }
          }
        })
      }
    },
    //  选中某一行
    handleSelectRow (selection,row) {
      this.selectedItems.add(row)
    },
    //  取消某一行
    handleCancelRow (selection,row) {
      for (let item of this.selectedItems) {
        if (item.id === row.id){
          this.selectedItems.delete(item)
        }
      }
    },
    setChecked() {
      let objData = this.$refs.purchaseTable.objData
      for (let index in objData) {
        for (let item of this.selectedItems) {
          if (item.id === objData[index].id) {
            objData[index]._isChecked = true
            break;
          }
        }
      }
    },
    submit(){
      console.log([...this.selectedItems]);
    }
  },
}
</script>

    reserve-selection
    仅对 type=selection 的列有效，类型为 Boolean，
    为 true 则会在数据更新之后保留之前选中的数据（需指定 row-key）
    Boolean — false
