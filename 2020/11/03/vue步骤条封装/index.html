    Step

    <template>
  <div class="step" :class="[{ 'step-vertical': !directionType },{ 'step-horizontal': directionType },status,statusType]">
    <i class="iconfont" :class="icon"></i>
    <span class="title" v-if="title">{{title}}</span>
    <span class="content" style="margin-left: 10px" v-if="content">{{content}}</span>
  </div>
</template>

<script>
export default {
  name: 'Step',
  // 监听current改变，将当前的current的状态设置为status:'process'
  watch: {
    current: {
      handler (newValue, oldValue) {
        this.changeCurrentStepsStatus(newValue, oldValue)
      },
      immediate: true
    }
  },
  computed: {
    directionType () {
      return this.direction === 'horizontal'
    },
    statusType () {
      return this.status + '-' + this.direction
    }
  },
  props: {
    // 每一步的状态属性,'process':'error':'wait':'finish'
    status: {
      type: String,
      default: 'process'
    },
    // 每一步的标题
    title: {
      type: String,
      default: ''
    },
    // 每一步的内容
    content: {
      type: String,
      default: ''
    },
    // 当前的current
    current: {
      type: Number,
      default: 0
    },
    // 是否自定义每一个步骤的status属性，如果自定义修改必须设置为true
    customStatus: {
      type: Boolean,
      default: false
    },
    // 图标属性，为iconfont直接是图标名称即可
    icon: {
      type: String,
      default: ''
    },
    // 方向 horizontal：vertical
    direction: {
      type: String,
      default: 'horizontal'
    }
  },
  methods: {
    changeCurrentStepsStatus (currentIndex, lastIndex) {
      this.$nextTick(() => {
        this.$parent.$parent.$parent.stepsContent.forEach((item, i) => {
          // 改变当前步骤的状态为 process
          if (i === currentIndex) {
            item.status = 'process'
          }
          // 默认状态时候使用
          if (!this.customStatus) {
            if (i === lastIndex) {
              if (lastIndex > currentIndex) {
                item.status = 'wait'
              } else {
                item.status = 'finish'
              }
            }
          }
        })
      })
    }
  }
}
</script>

next () {
  this.current += 1
  this.changeOtherStepsStatus()
},
prev () {
  this.current -= 1
  this.changeOtherStepsStatus()
},
changeOtherStepsStatus () {
  const index = this.current
  if (index === 1) {
    this.getAllCarPosition()
  }
  this.stepsContent.forEach((item, i) => {
    if (i === 0) {
      if (this.tableData.length) {
        item.status = 'finish'
      } else {
        item.status = 'error'
      }
    }
    if (i === 1) {
      if (this.selPosForm.position.length) {
        item.status = 'finish'
      } else {
        item.status = 'error'
      }
    }
    if (i > index) {
      item.status = 'wait'
    }
  })
}
