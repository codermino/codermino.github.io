    vue基于view-design封装的底部按钮组件
    2020/11/03 vue3 vue基于view-design封装的底部按钮组件 

    <template>
  <div class="bottomButton">
    <div class="bottom-button" :class="{'margin-top-20':hasMargin}" style="text-align: right;">
      <Button class="create-button margin-right-10" style="background-color:#525A6F !important;" @click="createApp">{{confirmButtonName}}</Button>
      <Button class="cancle-button" @click="cancel">{{cancelButtonName}}</Button>
    </div>
  </div>
</template>

<script>
export default {
  name: 'bottomButton',
  props: {
    confirmButtonName: {
      type: String,
      default: '创建'
    },
    cancelButtonName: {
      type: String,
      default: '取消'
    },
    hasMargin: {
      type: Boolean,
      default: true
    }
  },
  methods: {
    createApp () {
      this.$emit('clickCreate')
    },
    cancel () {
      this.$emit('clickCancle')
    }
  }
}
</script>

<style scoped lang="less">
.create-button{
  border-color: #525A6F !important;
  background-color: #525A6F !important;
  color: #fff !important;
  &:hover{
    color: #fff !important;
    border-color: #525A6F !important;
  }
}
 .cancle-button{
   background-color: #f7f7f7 !important;
   color: #333 !important;
   &:hover{
     color: #333 !important;
     border-color: #dcdee2 !important;
     background-color: #f7f7f7 !important;
   }
 }
