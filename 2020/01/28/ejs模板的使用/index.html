    ejs模板的使用
    2020/01/28 nodejs ejs模板的基本使用

    安装

    npm install ejs

    基本使用1

    //引入ejs
const ejs = require('ejs');
//定义字符串
let str = '今天是星期<%= a%>';
//定义data
data = {
  a:'天'
};
//ejs渲染str和data
let html = ejs.render(str, data);
//控制台输出
console.log(html);

    基本使用2

    1.首先创建一个ejs文件:index.ejs
<!doctype html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport"
          content="width=device-width, user-scalable=no, initial-scale=1.0, maximum-scale=1.0, minimum-scale=1.0">

</head>

</html>

2.使用ejs模板传数据和进行渲染
//引入fs和ejs
const fs = require('fs');
const ejs = require('ejs');
//读取文件(读取当前路径下的index.ejs文件)
fs.readFile('./index.ejs',function (err, data) {
  if(err){
    console.log(err);
  }else{
  //需要将模板转为字符串,因为在传输过程中是byte流
    const template = data.toString();
    const dictionary = {
      a:'天'
    };
    //使用ejs进行渲染，传入的数据有data,和ejs界面需要使用的数据
    const html = ejs.render(template, dictionary);
    //控制台输出渲染之后的结果
    console.log(html);
  }
});

    基本使用3

    //引入fs、ejs和http
const fs = require('fs');
const ejs = require('ejs');
const http = require('http');
//使用http开启一个服务器
let server = http.createServer(function (req, res) {
  fs.readFile('./index.ejs',function (err, data) {
    if(err){
      console.log(err);
    }else{
      const template = data.toString();
      const dictionary = {
        a:'天'
      };
      const html = ejs.render(template, dictionary);
      //显示
      res.writeHead(200,{"Content-Type":"text/html;charset=UTF8"});
      res.end(html)
    }
  });
});
//在3000端口监听
server.listen(3000, function () {
  console.log('服务器启动成功');
});
