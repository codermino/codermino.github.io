    vue路由调转传递参数
    2020/09/15 vue vue路由调转传递参数

    第一种方式

    todetail(id) {
  this.$router.push({
    path: `/detail/${id}`,
  })
}

// 第一种方式传参
{
    path: '/detail/:id',
    name: 'detail',
    meta:{
      title:'新闻详情'
    },
    component: () => import(/* webpackChunkName: "about" */ '@/components/detail/detail.vue')
}

接收
// this.id = this.$route.params.id;

这种形式在浏览器中是/detail/123的形式
刷新页面之后传递的参数不丢失

    第二种方式

    //第二种方式
todetail(id){
  this.$router.push({
    name: 'detail',
    params: {
      id: id
    }
  })
}

//第二种方式
{
    path: '/detail',
    name: 'detail',
    meta:{
      title:'新闻详情'
    },
    component: () => import(/* webpackChunkName: "about" */ '@/components/detail/detail.vue')
}

接收
// this.id = this.$route.params.id;

这种形式在浏览器中是/detail的形式
刷新页面之后传递的参数丢失

    第三种方式

    //第三种方式
todetail(id){
  this.$router.push({
    path: '/detail',
    query: {
      id: id
    }
  })
}

{
    path: '/detail',
    name: 'detail',
    meta:{
      title:'新闻详情'
    },
    component: () => import(/* webpackChunkName: "about" */ '@/components/detail/detail.vue')
}

//接收
this.id = this.$route.query.id;

这种形式在浏览器中是/detail?id=123的形式
刷新页面之后传递的参数不丢失
