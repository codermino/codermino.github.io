    <script>
      const app = Vue.createApp({
        data() {
          return {
            show: false,
          };
        },

        methods: {



          handleClick() {
            this.show = !this.show;
          },

          // 这里的钩子函数接收一个参数el，表示当前transition包裹的元素
          handleBeforeEnter(el){
            el.style.color = "red";
          },
          
          // 这里的钩子函数接收第一个参数el，表示当前transition包裹的元素
          // 这里的钩子函数接收第二个参数done，表示过程进行完成，如果没有则会一直处于过程中的状态
          handleEnterActive(el, done){
            const animation =  setInterval(() => {
                const color = el.style.color;

                if(color === "red"){
                    el.style.color = "green";
                }else{
                    el.style.color = "red";
                }
              },1000)

              setTimeout(() => {
                  clearInterval(animation);
                  done();
              }
              ,4000)
          },


          handleEnterEnd(){
            alert(666);
          }
        

          
        },

        // 可以指定type是transition还是animation来确定以哪个为基准。
        // type='transition'以transition为基准,transition结束那么animation也结束
        // 同理type='animation'一样

        // 如果设置了duration，那么transition和animation的时间都失效。单位为ms
        // 也可以分别设置入场和出场的时间:duration='{enter:1000,leave:2000}'
        // <transition type='transition'>
        // <transition type='transition' :duration='1000'>
        // template: `
        //       <transition type='transition' :duration='{enter:1000,leave:2000}'>
        //           <div v-show="show">hello world</div>
        //        </transition> 
        //           <button @click="handleClick">切换</button>  
        //        `,

        // template: `
        //        <transition :duration="{enter:1000, leave:3000}">
        //           <div v-show="show">hello world</div>
        //        </transition> 
        //           <button @click="handleClick">切换</button>  
        //        `,

        // 使用Js控制过渡和动画，需要将css禁止掉，css='false'
        template: `
                <transition
                    :css="false"
                    @before-enter="handleBeforeEnter"  
                    @enter="handleEnterActive"  
                    @after-enter="handleEnterEnd" 
                    @before-leave=""
                    @leave = ""
                    @after-leave =""
                    
                >
                   <div v-show="show">hello world</div>
                </transition> 
                   <button @click="handleClick">切换</button>  
                `,
      });

      const vm = app.mount("#root");
    </script>
