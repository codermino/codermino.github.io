    在render函数中使用slot

    需求：
    既能够根据key来创建对应的dom,
    也能够根据slot，然后写对应的具名slot进行渲染

    父组件

    <template>
  <div class="container">
    <render-template :level="4" :tableData="renderSonData">
<!--      <div class="jah" slot="myslot">render function</div>-->
<!--      <div class="jah" slot="myslot2">render function2</div>-->
      <div class="jah" slot="address">render function2</div>
      <div class="jah" slot="phone">render function3</div>
    </render-template>
  </div>
</template>

<script>
import renderTemplate from './slotsTestSon.vue'
export default {
  name: "slotsTest",
  components:{
    renderTemplate
  },
  data() {
    return {
      num: 123,
      renderSonData:[
        {
          title: 'Name',
          key: 'name'
        },
        {
          title: 'Age',
          key: 'age'
        },
        {
          title: 'Address',
          slot: 'address',
        },
        {
          title: 'phone',
          slot: 'phone',
        }
      ]
    }
  },
}
</script>

    子组件

    <script>
export default {
  name: "slotsTestSon",
  props: {
    level: {
      type: Number,
      required: true
    },
    tableData: {
      type: Array,
      required: true
    }
  },
  render: function (createElement) {
    return createElement(
      // 'h' + this.level, {
      'div', {
        'class': {
          show: true,
          hide: false,
        },
        style: {
          width: '200px',
          height: '400px',
          background: 'red',
        },
        attrs: {
          name: 'h-ex',
          id: 'h-id'
        },
        props: {
          myprops: true,
        },
        on: {
          click:  (event)=> {
            alert(this.level)
          }
        },
      },
      // 遍历tableData数组
      this.tableData.map((item)=>{
        if (!item.slot && item.key){
          return createElement('span',
            {
              domProps:{
                innerHTML:item.title
              },
              style: {
                display:'block',
                backgroundColor:'blue'
              }
            })
        } else {
          // 这样的形式是直接创建一个slot对应的内容
          const slotName = item.slot
          return this.$slots[slotName]
          // 这样的形式是将slot对应的内容包裹在一个div中
          // return createElement('div',
          //   [this.$slots[slotName]])
        }
      })
      // [
      //    this.$slots.myslot,
      //    this.$slots.myslot2,
      //   createElement('div', {
      //     domProps: {
      //       innerHTML: 'holle render'
      //     }
      //   })
      // ]
    )
  },
}
</script>
