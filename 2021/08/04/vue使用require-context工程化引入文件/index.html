    vue使用require.context工程化引入文件
    2021/08/04 vue vue使用require.context工程化引入文件

    自动注册全局组件

    import vue from 'vue'

const requireComponent = require.context( // webpack的api，一个函数方法，匹配文件
    './common', // 查看当前目录下的文件(查找需要文件的相对路径)
    true, // 是否查看子文件
    /\.vue$/ // 匹配方式正则表达式，只查看后缀为.vue的文件
)
// 循环获取到的文件，可在循环时处理文件名
requireComponent.keys().forEach((fileName) => {
    // 获取组件配置(组件模板)
    const componentConfig = requireComponent(fileName)
    // 将被注册的组件名字,对获取的组件文件名进行处理
    const componentName = fileName
        .split('\/')[fileName.split('\/').length - 1]
        .replace(/^\.\/_/, '')
        .replace(/\.\w+$/, '')
        .split('./')
        .join('')
    // 将组件在循环中注册到全局，
    vue.component(componentName, // 依据文件名处理好的，将要被注册到全局的组件名
        componentConfig.default || componentConfig)
})

    自动加载css|less文件

    const requireComponent = require.context(
    './style',
    true,
    /\.less$|\.css$/
)

const requireAll = requireContext => requireContext.keys().map(requireContext);
requireAll(requireComponent);
