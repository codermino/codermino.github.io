    前言

    用于图片浏览的Vue组件，支持旋转、缩放、翻转等操作，基于viewer.js。

    具体使用代码

    <template>
    <div class="vviewer">
        <!-- directive -->
        <div ref="images" class="images" v-viewer.rebuild="{inline: false, movable: false, navbar: 1, show:this.startShow}">
            <img v-for="src in images" :key="src" :src="src" @click="clickImg(src)">
        </div>
        <!-- component -->
<!--        <viewer :images="images">-->
<!--            <img v-for="src in images" :key="src" :src="src">-->
<!--        </viewer>-->
        <!-- api -->
        <button type="button" @click="show">展示</button>
<!--        <button type="button" @click="addPic">添加图片</button>-->


        <div v-viewer="options" class="images">
            <template v-for="(image,index) in images2">
                <img :src="image.thumbnail" class="image" :key="index"
                     :data-src="image.source"
                >
            </template>
        </div>
    </div>
</template>

<script>
    import 'viewerjs/dist/viewer.css'
    import VueViewer from 'v-viewer'
    import Vue from 'vue'
    Vue.use(VueViewer)
    export default {
        name: "vviewer",
        created () {
            const base = parseInt((Math.random() * 60), 10) + 10
            for (let i = 0; i < 5; i++) {
                this.images2.push({
                    thumbnail: `https://picsum.photos/id/${base + i}/300/200`,
                    source: `https://picsum.photos/id/${base + i}/600/400`
                })
            }
        },
        data() {
            return {
                images: [
                    "https://picsum.photos/200/200",
                    "https://picsum.photos/300/200",
                    "https://picsum.photos/250/200"
                ],
                options: {
                    viewed (e) {
                        console.log(e.type)
                        this.viewer.rotateTo(180)
                    },
                    show(e){
                        console.log(e.type)
                    }
                },
                images2: []
            }
        },
        methods: {
            startShow(e){
                console.log(e)
            },
            show() {
                // this.$viewerApi({
                //     images: this.images,
                // })
                const viewer = this.$refs.images.$viewer
                viewer.show()
            },
            // addPic () {
            //     this.images.push("https://picsum.photos/250/100")
            // }
            clickImg(src){
                console.log(src)
            }
        }
    }
</script>
