    ## 直接使用

    <template>
    <div class="roll">
        <div class="cont" ref='cont'>
            <div class="contlist" ref='contlist' @mouseover="stopScroll" @mouseout="startScroll">
                <ul>
                    <li v-for="(item, index) in list" :key="index">
                        {{ index + 1 }} 、<em>{{ item.name }}</em> {{ item.time }}
                    </li>
                </ul>
            </div>
        </div>
    </div>
</template>

<script>
    export default {
        name: "roll",
        data() {
            return {
                list: [
                    {name: '负限位报警', time: '2019-12-18 16:20:34'}
                ],
                contentWidth: 0,
                scrollW: 0,
                flag: true // true代表目前是right负的，false代表是正的
            }
        },
        methods: {
            //鼠标悬停，停止滚动
            stopScroll () {
                clearInterval(this.scrollTime)
            },
            // //鼠标移开 ，接着滚动
            startScroll () {
                var target = this.$refs.contlist;
                const currentRight = target.style.right.split('px')[0]
                // right为正数的情况
                if (currentRight.indexOf('-') === -1){
                    this.flag = false
                    this.scrollW = Number(currentRight)
                } else {
                    // right为负数的情况
                    this.flag = true
                    let temp = currentRight.split('-')[currentRight.split('-').length - 1]
                    this.scrollW = Number(temp)
                }
                this.scroll()
            },
            scroll (){
                var initRight = this.flag ? -this.scrollW : this.scrollW;
                const target = this.$refs.contlist
                //定时器
                this.scrollTime = setInterval(()=>{
                    initRight ++;
                    // if(initRight >= target.offsetWidth + this.contentWidth ){
                    if(initRight >= this.contentWidth ){
                        initRight =  -target.offsetWidth;
                    }
                    target.style.right = initRight +'px';//获取不到translateX的值改用 left
                },10)
            }
        },
        mounted() {
            //实例挂载完毕前
            //调用滚动代码
            //整个大的整体的宽度
            this.contentWidth = this.$refs.cont.offsetWidth
            // 滚动内容的宽度
            const target = this.$refs.contlist
            this.scrollW = target.offsetWidth
            // 让滚动的内容在整个容器的最右边
            target.style.right = -this.scrollW +'px';
            this.scroll()
        }
    }
</script>

    }

        height: 20px;
