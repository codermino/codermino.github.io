    html的iframe页面

    <!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>iframetest</title>
</head>
<body>
<script>
  //监听信息
  window.addEventListener("message",handleMessage);
  function handleMessage (data) {
    console.log(data)
  }
  //传递信息
  // window.parent.postMessage('123123123token', '*');
  function sendMessageToFather () {
    window.parent.postMessage({
      msg: 'son father'
    }, '*');
  }
</script>
<div class="iframetest">
    <button onclick="sendMessageToFather()">向父亲传递消息</button>
</div>
</body>
</html>

    vue页面使用iframe,并进行通信

    <template>
    <div class="iframeprac">
      <el-button @click="sendMessage1">消息1</el-button>
      <iframe style="width: 100%;height: 100vh;"
              src="static/iframetest.html"
              id="iframe"
              class="iframe">
      </iframe>
    </div>
</template>

<script>
const IFRAME_HOST = 'http://localhost:8080'
export default {
  name: 'iframeprac',
  mounted () {
    window.addEventListener('message', e => {
      if (e && e.data && e.origin === IFRAME_HOST) {
        if (typeof e.data === 'string') {
          console.log('message===============', e)
          if (e.data === 'token') {
            const token = {
              type: 'token',
              value: localStorage.getItem('token')
            }
            document.getElementById('iframe').contentWindow.postMessage(token, IFRAME_HOST)
            return
          }
          if (e.data === 'hysInfo') {
            console.log('hysInfo')
            return
          }
          if (e.data.includes('Board')) {
            console.log('Board')
          }
        } else if (typeof e.data === 'object') {
          console.log(e.data)
        }
      }
    })
  },
  methods: {
    sendMessage1 () {
      document.getElementById('myIframe').contentWindow.postMessage({
        type: 'message-1',
        body: {
          args: '消息内容 可以是任何对象'
        }
      }, IFRAME_HOST)
    }
  }
}
</script>
