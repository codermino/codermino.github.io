    <template>
    <div class="tableDrag">
      <el-table
        id="el-table-print"
        stripe
        :data="tableData"
        border
        align="center"
        :header-cell-style="{background: '#F2F2F2', color: '#555'}"
        ref="goodsDetailTable"
        row-key="id"
        @sort-change="sortChange"
      >
        <template v-for="(item, index) in dropCol">
          <el-table-column
            :key="index"
            :prop="item.prop"
            :label="item.label"
            :min-width="item.width"
            :sortable="item.sortable"
            :show-overflow-tooltip="item.collapse"
          >
            <template slot-scope="scope">
              <slot :name="`lineInsert_${item.prop}`" :row="scope.row">{{ scope.row[item.prop] }}</slot>

<script>
import Sortable from 'sortablejs'
export default {
  name: 'tableDrag',
  mounted () {
    this.$nextTick(() => {
      this.$refs.goodsDetailTable.doLayout()
      this.changeCursor()
      this.addThListener()
    })
  },
  destroyed () {
    this.reThListener()
  },
  data () {
    return {
      dropCol: [
        {
          prop: 'num',
          label: '序号',
          insertFlagArr: {},
          collapse: false
        },
        {
          prop: 'applyFileTitle',
          width: 367,
          label: '标题',
          sort: '',
          sortable: 'custom',
          insertFlagArr: {
            alertbutton: false
          },
          collapse: true
        },
        {
          prop: 'dealDay',
          width: 155,
          label: '停留时间',
          sort: '',
          sortable: 'custom',
          insertFlagArr: {
            alertbutton: false
          },
          collapse: false
        },
        {
          prop: 'itemName',
          width: 165,
          label: '类型',
          sort: 'ascending',
          sortable: 'custom',
          insertFlagArr: {
            alertbutton: false
          },
          collapse: false
        },
        {
          prop: 'currentStatus',
          width: 138,
          label: '状态',
          sort: '',
          sortable: false,
          insertFlagArr: {
            alertbutton: false
          },
          collapse: false
        },
        {
          prop: 'applyFileNum',
          width: 135,
          label: '文号',
          sort: '',
          sortable: false,
          insertFlagArr: {
            alertbutton: '收起'
          },
          collapse: false
        },
        {
          prop: 'applicantName',
          width: 238,
          label: '名称',
          sort: '',
          sortable: 'custom',
          insertFlagArr: {
            alertbutton: false
          },
          collapse: false
        },
        {
          prop: 'remnant',
          width: 100,
          label: '剩余时间',
          sort: '',
          sortable: false,
          insertFlagArr: {
            alertbutton: false
          },

        },
        {
          prop: 'applyTime',
          width: 200,
          label: '申请时间',
          sort: '',
          sortable: false,
          insertFlagArr: {
            alertbutton: false
          },
          collapse: false
        },
        {
          prop: 'preProcessorName',
          width: 180,
          label: '上一人',
          sort: '',
          sortable: 'custom',
          insertFlagArr: {
            alertbutton: false
          },
          collapse: false
        },
        {
          prop: 'dispatchNum',
          width: 200,
          label: '文号',
          sort: '',
          sortable: false,
          insertFlagArr: {
            alertbutton: false
          },
          collapse: false
        },
        {
          prop: 'dispatchTitle',
          width: 150,
          label: '标题',
          sort: '',
          sortable: false,
          insertFlagArr: {
            alertbutton: false
          },
          collapse: false
        }
      ],
      tableData: [{
        num: '123',
        applyFileTitle: '上海市普陀区金沙江路 1518 弄',
        dealDay: '123',
        itemName: '123',
        currentStatus: '123',
        applyFileNum: '123',
        applicantName: '123',
        remnant: '123',
        applyTime: '123',
        preProcessorName: '123',
        dispatchNum: '123',
        dispatchTitle: '123'
      }]
    }
  },
  methods: {
    // 鼠标样式,拖拽时move,改变列宽时col-resize
    changeCursor () {
      const leafArr = document.querySelectorAll('.is-leaf')
      for (let i = 0; i < leafArr.length - 1; i++) {
        leafArr[i].onmousemove = ev => {
          leafArr[i].style.cursor = ''
          if (ev.offsetX > leafArr[i].offsetWidth - 3) {
            leafArr[i].style.cursor = 'col-resize'
          } else {
            leafArr[i].style.cursor = 'move'
          }
        }
      }
    },
    // 行拖拽
    rowDrop () {
      const tbody = document.querySelector('.el-table__body-wrapper tbody')
      const _this = this
      Sortable.create(tbody, {
        onEnd ({ newIndex, oldIndex }) {
          const currRow = _this.tableData.splice(oldIndex, 1)[0]
          _this.tableData.splice(newIndex, 0, currRow)
        }
      })
    },
    // 添加监听器
    addThListener () {
      const thArr = document.querySelectorAll('.el-table__header-wrapper th')
      for (let i = 1; i < thArr.length - 2; i++) {
        const element = thArr[i].childNodes[0]
        if (element.addEventListener) {
          // 所有主流浏览器, 除了 IE 8 及更早 IE 版本
          element.addEventListener('dragover', this.allowDrop)
          element.addEventListener('drop', this.drop)
          element.addEventListener('dragstart', this.drag)
        } else if (element.attachEvent) {
          // IE 8 及更早 IE 版本
          element.attachEvent('dragover', this.allowDrop)
          element.attachEvent('drop', this.drop)
          element.attachEvent('dragstart', this.drag)
        }
        element.setAttribute('draggable', true)
        this.addName(element, i)
      }
    },
    // 递归给元素添加属性名
    addName (element, i) {
      element.setAttribute('name', i)
      if (element.children.length > 0) {
        const childNodeArr = element.childNodes
        for (let j = 0; j < childNodeArr.length; j++) {
          if (childNodeArr[j].nodeType === 1) {
            childNodeArr[j].setAttribute('name', i)
          }
          if (childNodeArr[j].hasChildNodes()) {
            this.addName(childNodeArr[j], i)
          }
        }
      }
    },
    // 移除监听器
    reThListener () {
      const thArr = document.querySelectorAll('th')
      for (let i = 1; i < thArr.length - 2; i++) {
        const element = thArr[i].childNodes[0]
        if (element.removeEventListener) {
          // 所有主流浏览器, 除了 IE 8 及更早 IE 版本
          element.removeEventListener('dragover', this.allowDrop)
          element.removeEventListener('drop', this.drop)
          element.removeEventListener('dragstart', this.drag)
        } else if (element.detachEvent) {
          // IE 8 及更早 IE 版本
          element.detachEvent('dragover', this.allowDrop)
          element.detachEvent('drop', this.drop)
          element.detachEvent('dragstart', this.drag)
        }
      }
    },
    drag (ev) {
      ev.dataTransfer.setData('Text', ev.target.getAttribute('name'))
    },
    allowDrop (ev) {
      ev.preventDefault()
    },
    drop (ev) {
      // 阻止默认行为
      ev.preventDefault()
      const oldIndex = ev.target.getAttribute('name')
      const newIndex = ev.dataTransfer.getData('Text')

      // 切换表头对应数据
      this.dropCol.splice(
        oldIndex,
        1,
        ...this.dropCol.splice(newIndex, 1, this.dropCol[oldIndex])
      )
      // 每回拖动ElementUI的自定义插槽会清空我的自定义name, 所以重新添加name
      this.$nextTick(() => {
        // 解决表头错位问题
        this.$refs.goodsDetailTable.doLayout()

        const eleArr = document.querySelectorAll('.el-table__header-wrapper th')
        for (let i = 0; i < eleArr.length - 2; i++) {
          const element = eleArr[i].childNodes[0]
          this.addName(element, i)
        }
      })
    },
    sortChange ({ column, prop, order }) {
      console.log(column, prop, order)
      this.dropCol.forEach(item => {
        if (item.prop === prop) {
          item.sort = order
        }
      })
    },
    buttonFu () {
      this.dropCol.forEach(item => {
        item.collapse = !item.collapse
      })
    }
  }
}
</script>
