    第一种iview+less换肤

    具体项目代码参考自己的github或者码云
    这种方式是借助less的less.modifyVars()方法进行颜色的更换
    需要在color.less中列出需要进行主题颜色更改的类并且绑定同一个变量
    vue.config.js文件

    module.exports = {
    css: {
        loaderOptions: {
            less: {
                javascriptEnabled: true,
                modifyVars: {
                    'my-primary-color': '#2264A0',
                }
            }
        }
    }
}

    scheme.js文件

    let lessNodesAppended;

const updateTheme = (primaryColor='#2264A0') => {
  console.log(primaryColor)
  // debugger;
  if (!primaryColor) {
    // return;
    primaryColor = '#2264A0';
  }
  if(primaryColor == 'null') primaryColor = '#2264A0';
  console.info(`正在编译主题!`)
  function buildIt() {
    // 正确的判定less是否已经加载less.modifyVars可用
    if (!window.less || !window.less.modifyVars) {
      return;
    }
    // less.modifyVars可用
    window.less.modifyVars({
      '@my-primary-color': primaryColor,
    })
      .then(() => {
        // console.log(`成功`);

        // this.$Message.success('换肤成功!');
      })
      .catch((reason) => {
        console.error(`失败${reason}`);
      });
  }
  if (!lessNodesAppended) {
    // insert less.js and color.less
    const lessStyleNode = document.createElement('link');
    const lessConfigNode = document.createElement('script');
    const lessScriptNode = document.createElement('script');
    lessStyleNode.setAttribute('rel', 'stylesheet/less');
    // 下方这个color.less位置大家也可以按需修改
    lessStyleNode.setAttribute('href', __webpack_public_path__ + 'static/color.less')
    lessConfigNode.innerHTML = `
      window.less = {
        async: true,
        env: 'production',
        javascriptEnabled: true
      };
    `;
    // less的src地址也可以是cdn地址
    lessScriptNode.src = '/static/less.min.js';
    lessScriptNode.async = true;
    lessScriptNode.onload = () => {
      buildIt();
      lessScriptNode.onload = null;
    };
    document.body.appendChild(lessStyleNode);
    document.body.appendChild(lessConfigNode);
    document.body.appendChild(lessScriptNode);
    lessNodesAppended = true;
  } else {
    buildIt();
  }
};
export { updateTheme }

    第二种element-ui+scss换肤(推荐)

    具体项目细节还是看自己的github和码云
    这种方式主要是基于webpack-theme-color-replacer
    这种方式不需要提前将element-ui的组件的class提前列出。
    可以直接修改element-ui的组件颜色等，同时也可以修改自定义的
    class颜色等，只需要将颜色改为主题颜色的scss变量即可。
    vue.config.js

    const ThemeColorReplacer = require('webpack-theme-color-replacer');
const forElementUI = require('webpack-theme-color-replacer/forElementUI');

module.exports = {
    // eslint-loader 是否在保存的时候检查
    lintOnSave: process.env.NODE_ENV === 'development',
    //生产环境是否生成 sourceMap 文件，一般情况不建议打开
    productionSourceMap: false,
    // webpack配置
    configureWebpack: {
        plugins: [
            new ThemeColorReplacer({
                fileName: 'css/theme-colors.[contenthash:8].css',
                matchColors: forElementUI.getElementUISeries('#2264A0'),
                changeSelector: forElementUI.changeSelector,
                // 防止history模式刷新丢失颜色
                injectCss:true
            })
        ]
    },
    css: {
        loaderOptions: {
            sass: {
                prependData: `@import "~@/styles/element-variables.scss";`
            }
        },
        // 启用 CSS modules
        // requireModuleExtension: false,
        // 是否使用 CSS 分离插件
        extract: true,
        // 开启 CSS source maps, 一般不建议开启
        sourceMap: false
    }
};

    theme.js

    import forElementUI from 'webpack-theme-color-replacer/forElementUI';
import client from 'webpack-theme-color-replacer/client';
import {Message} from 'element-ui'
import {LOCAL_STORAGE_KEY} from './const'
import Storage from './storage'
import variables from '../styles/element-variables.scss'
// import store from '@/store/index.js';

const theme = {
  get() {
    //return Storage.get(LOCAL_STORAGE_KEY.THEME_COLOR) || variables.theme
    // if(Storage.get(LOCAL_STORAGE_KEY.THEME_COLOR)){
    if(localStorage.getItem(LOCAL_STORAGE_KEY.THEME_COLOR)){
      // return JSON.parse(Storage.get(LOCAL_STORAGE_KEY.THEME_COLOR))[store.state.user.userInfo.userId] || variables.theme
      return variables.theme
    }else {
      return variables.theme
    }
  },
  // 初始化，设置主题颜色
  init() {
    const options = {
      newColors: forElementUI.getElementUISeries(this.get())
    };
    client.changer.changeColor(options, Promise)
  },
  // 修改主题颜色
  change(themeColor) {
    // let userId = store.state.user.userInfo.userId
    // let loclThemeColor = Storage.get(LOCAL_STORAGE_KEY.THEME_COLOR)
    // if(loclThemeColor){
    //   let themeColorObj = JSON.parse(loclThemeColor)
    //   themeColorObj[userId] = themeColor
    //   Storage.set(LOCAL_STORAGE_KEY.THEME_COLOR, JSON.stringify(themeColorObj))
    // }else {
    //   let i = { [userId]: themeColor }
    //   Storage.set(LOCAL_STORAGE_KEY.THEME_COLOR, JSON.stringify(i))
    // }
    localStorage.removeItem(LOCAL_STORAGE_KEY.THEME_COLOR)
    localStorage.setItem(LOCAL_STORAGE_KEY.THEME_COLOR, themeColor)
    const options = {
      newColors: forElementUI.getElementUISeries(themeColor)
    };
    client.changer.changeColor(options, Promise).finally(() => Message({
      message: '换肤成功',
      type: 'success'
    }))
  }
};

export default theme;
