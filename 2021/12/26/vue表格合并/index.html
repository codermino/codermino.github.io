    el-table表格合并

    公共方法

    export function getSpanArr (data, field) {
  const spanArr = []
  let pos = 0
  for (var i = 0; i < data.length; i++) {
    if (i === 0) {
      spanArr.push(1)
      pos = 0
    } else {
      if (data[i][field] === data[i - 1][field]) {
        spanArr[pos] += 1
        spanArr.push(0)
      } else {
        spanArr.push(1)
        pos = i
      }
    }
  }
  return spanArr
}

/*
    labelArr 合并字段Array
    field 参照合并的字段
    如果field对应的字段相同，那么九江labelArr数组中所对应的
    表格数据合并。
*/

export function objectSpanMethod ({ column, rowIndex }, data, field, labelArr) {
  if (labelArr.includes(column.label)) {
    const _row = getSpanArr(data, field)[rowIndex]
    const _col = _row > 0 ? 1 : 0
    return {
      rowspan: _row,
      colspan: _col
    }
  }
}
