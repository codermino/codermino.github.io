    后端返回文件流，前端进行下载

    封装一个工具js文件和工具函数

    // 导入axios
import axios from 'axios'
import { url, productionBaseUrl } from './ajaxUrl'
const env = window.$env ? window.$env : process.env.NODE_ENV

const util = {}

util.baseUrl = env === 'development' ? url : productionBaseUrl

util.http = axios.create({
  baseURL: util.baseUrl,
  timeout: 50000
})

util.ajaxWithFile = function (url, method, params) {
  return new Promise((resolver, reject) => {
    var res = resolver
    util.http({
      baseURL: util.baseUrl,
      method: method,
      url: url,
      data: params,
      responseType: 'blob'
    }).then(response => {
      res(response)
    }).catch((error) => {
      reject(error)
    })
  })
}
util.downloadDoc = function (data, fileName) {
  var blob = new Blob([data])
  if (window.navigator.msSaveBlob) {
    try {
      // window.navigator.msSaveBlob(blob, fileName+'.doc')
      window.navigator.msSaveBlob(blob, fileName)
    } catch (e) {
      console.log(e)
    }
  } else {
    const url = window.URL.createObjectURL(blob)
    const link = document.createElement('a')
    link.style.display = 'none'
    link.href = url
    link.setAttribute('download', fileName)
    document.body.appendChild(link)
    link.click()
  }
}

export default util

Vue.prototype.$http = util.http

new Vue({
  router,
  render: h => h(App)
}).$mount('#app')

    runAccessory (datas) {
  const params = {
    newsId: datas.newsId
  }
  this.$util.ajaxWithFile('', 'post', params).then(res => {
    if (res.headers['content-disposition']) {
      const fileName = decodeURI(res.headers['content-disposition'].split(';')[1].split('=')[1])
      // rea.data是后端返回来的流数据
      this.$util.downloadDoc(res.data, fileName)
    }
  })
}
