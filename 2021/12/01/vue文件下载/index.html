    需求和原理

    调用一个接口，后端返回的是stream数据
    接收到数据之后装成二进制，然后转成url，并且进行下载

    封装一个工具方法

    util.ajaxWithFile = function (url, method, params) {
  return new Promise((resolver, reject) => {
    var res = resolver
    util.http({
      baseURL: util.baseUrl,
      method: method,
      url: url,
      data: params,
      responseType: 'blob'
    }).then(response => {
      res(response)
    }).catch((error) => {
      reject(error)
    })
  })
}
util.downloadDoc = function (data, fileName) {
  var blob = new Blob([data])
  if (window.navigator.msSaveBlob) {
    try {
      // window.navigator.msSaveBlob(blob, fileName+'.doc')
      window.navigator.msSaveBlob(blob, fileName)
    } catch (e) {
      console.log(e)
    }
  } else {
    const url = window.URL.createObjectURL(blob)
    const link = document.createElement('a')
    link.style.display = 'none'
    link.href = url
    link.setAttribute('download', fileName)
    document.body.appendChild(link)
    link.click()
  }
}

    Vue.prototype.$util = util

    使用方式

    this.$util.ajaxWithFile('/api/webFiles/selectApps', 'post', params).then(res => {
    const fileName = decodeURI(res.headers['content-disposition'].split(';')[1].split('=')[1])
    this.$util.downloadDoc(res.data, fileName)
  })
