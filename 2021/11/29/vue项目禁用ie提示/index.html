    vue项目禁用ie提示
    2021/11/29 vue vue项目禁用ie提示

    index.html文件

    <!DOCTYPE html>
<html lang="">
<head>
  <meta charset="utf-8">
  <meta http-equiv="X-UA-Compatible" content="IE=edge">
  <meta name="viewport" content="width=device-width,initial-scale=1.0">
  <link rel="icon" href="<%= BASE_URL %>favicon.ico">
  <title><%= htmlWebpackPlugin.options.title %></title>
</head>
<body>
<noscript>
  <strong>123</strong>
</noscript>
<div id="app"></div>
<iframe id='IEIframe' style='display: none;' width="100%" height="700px" src="/static/ie.html" frameborder="0"></iframe>
<script type="text/javascript">
    /* 如果是ie浏览器 */
  if (!!window.ActiveXObject || "ActiveXObject" in window) {
    document.querySelector('#IEIframe').style.display = 'block'
  }
</script>
<!-- built files will be auto injected -->
</body>
</html>

    App.vue文件

    <template>
  <div id="app" v-if='!isShow'>
    <router-view/>
  </div>
</template>

<script>
export default {
  data () {
    return {
      isShow: false
    }
  },
  mounted () {
    this.isShow = !!window.ActiveXObject || 'ActiveXObject' in window
  }
}
</script>
