    请求拦截和响应拦截

    import axios from 'axios'
import vm from '../main'

const extraParams = {
  test: '123'
}

function requestMethod (config) {
  let i = ''
  switch (config.method) {
    case 'get':
      i = config.params ? config.params : {}
      config.params = Object.assign(i, extraParams)
      break
    case 'post':
      i = config.data ? config.data : {}
      config.data = Object.assign(i, extraParams)
      break
    case 'put':
      i = config.data ? config.data : {}
      config.data = Object.assign(i, extraParams)
      break
    case 'delete':
      i = config.params ? config.params : {}
      config.params = Object.assign(i, extraParams)
      break
  }
}

axios.interceptors.request.use(function (config) {
  config.headers.Authorization = localStorage.token
  requestMethod(config)
  return config
}, function (error) {
  // 请求失败的处理
  return Promise.reject(error)
})

// http响应拦截
axios.interceptors.response.use(function (res) {
  return res
}, function (error) {
  if (error.response) {
    switch (error.response.status) {
      case 401:
        if (vm.$route.name !== 'login') {
          vm.$Message.warning({
            content: '身份认证信息已失效，请重新登录。',
            duration: 2,
            onClose: () => {
              vm.$router.push({ name: 'login' })
            }
          })
        } else {
          return Promise.reject(error)
        }
        break
      case 403:
        vm.$Message.warning({
          content: error.response.data.message,
          duration: 3
        })
        break
      default:
        return Promise.reject(error)
    }
  } else if (error.message) {
    if (error.message === 'Network Error') {
      vm.$Message.error('网络连接错误，请检查网络环境！')
    } else if (error.message.indexOf('timeout') >= 0) {
      vm.$Message.error('网络连接超时，请重试！')
    } else {
      vm.$Message.error(error.message)
    }
  } else {
    vm.$Message.error('知识盲区的错误，联系程序员小哥哥吧！')
  }
})

export default axios
