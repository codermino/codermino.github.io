    非动态加载的tree和Table的联动

    左侧是tree右侧是Table
    点击左侧的tree,右边的表格加载的是对应的tree节点下面子节点的信息
    右侧新增，修改和删除会修改Table中的数据，同时也会影响到左侧tree

    <template>
  <div class="appTypeManage" style="display: flex;flex-direction: column;height: 100%;">
    <Modal
      title="删除数据"
      :mask-closable="false"
      v-model="showModal"
      @on-ok="confirmDel"
      @on-cancel="cancelDel"
    >
      <p style="color: red;">确认删除该条信息?</p>
    </Modal>
    <Modal
      v-model="addmodalshow"
      :title="title"
      :mask-closable="false">
      <Form :model="editformData" :label-width="80">
        <FormItem label="分类名称">
          <Input v-model="editformData.catalogname"></Input>
        </FormItem>
        <FormItem label="备注">
          <Input v-model="editformData.remarks" type="textarea"></Input>
        </FormItem>
        <FormItem label="排序字段">
          <Input v-model="editformData.ordernum" type="number"></Input>
        </FormItem>
      </Form>
      <div slot="footer">
        <bottom-button :confirmButtonName="confirmButtonName" :hasMargin="false" @clickCreate="ok" @clickCancle="cancel"></bottom-button>
      </div>
    </Modal>
    <div class="container">
      <Card class="card-tree">
        <el-tree
          @node-click="clickNode"
          ref="tree"
          :props="props"
          node-key="id"
          :expand-on-click-node="true"
          :default-expanded-keys="defaultExpandsId"
          :data="treedata">
          <span class="custom-tree-node" slot-scope="{ node, data }">
            {{data.catalogname}}
          </span>
        </el-tree>
      </Card>
      <Card class="table-content">
        <Row style="text-align: right" class="margin-bottom-10">
          <Button type="primary" class="blue-button" @click="addApp">添加应用</Button>
        </Row>
        <Table :columns="columns1" :data="tableData.slice((currentPage-1)*5,5*currentPage)">
          <template slot-scope="{ row, index }" slot="action">
            <Button class="blue-button" type="primary" size="small" style="margin-right: 5px" @click="edit(row, index)">编辑</Button>
            <Button class="red-button" type="primary" size="small" @click="deleteRow(row,index)">删除</Button>
          </template>
        </Table>
        <Page style="text-align: right;" class="margin-top-10" :total="tableData.length" :page-size="5" @on-change="pageChange"/>
      </Card>
    </div>
  </div>
</template>

<script>
export default {
  name: 'appTypeManage',
  data () {
    return {
      props: {
        label: 'catalogname',
        children: 'children'
      },
      treedata: [
        {
          id: '1',
          catalogname: '一级 1',
          children: [{
            id: '1-1',
            catalogname: '二级 1-1',
            children: [{
              id: '1-1-1',
              catalogname: '三级 1-1-1'
            }]
          }]
        }, {
          id: '2',
          catalogname: '一级 2',
          children: [{
            id: '2-1',
            catalogname: '二级 2-1',
            children: [{
              id: '2-1-1',
              catalogname: '三级 2-1-1'
            }]
          }, {
            id: '2-2',
            catalogname: '二级 2-2',
            children: [{
              id: '2-2-1',
              catalogname: '三级 2-2-1'
            }]
          }]
        }, {
          id: '3',
          catalogname: '一级 3',
          children: [{
            id: '3-1',
            catalogname: '二级 3-1',
            children: [{
              id: '3-1-1',
              catalogname: '三级 3-1-1'
            }]
          }, {
            id: '3-2',
            catalogname: '二级 3-2',
            children: [{
              id: '3-2-1',
              catalogname: '三级 3-2-1'
            }]
          }]
        }],
      defaultExpandsId: [],
      columns1: [
        {
          type: 'index',
          title: '序号',
          align: 'center'
        },
        {
          title: '分类名称',
          key: 'catalogname',
          align: 'center'
        },
        {
          title: '备注',
          key: 'remarks',
          align: 'center'
        },
        {
          title: '操作',
          slot: 'action',
          width: 150,
          align: 'center'
        }
      ],
      tableData: [],
      showModal: false,
      selectRow: {},
      nodeList: [],
      treeNodeIndex: -1,
      currentNode: {},
      addmodalshow: false,
      formType: 'add',
      editformData: {
        id: Math.ceil(Math.random() * 10 + 10),
        catalogname: null
      },
      currentPage: 1,
      title: '添加应用',
      confirmButtonName: '创建'
    }
  },
  components: {
    bottomButton: () => import('../components/bottom-button/bottomButton')
  },
  methods: {
    loadNode (node, resolve) {
      if (node.level === 0) {
        let root = {}
        this.$http.get('/api/getDmpAppCatalogsByParentId/0').then(response => {
          root = response.data.data[0]
          this.defaultExpandsId.push(root.id)
          resolve([root])
        }).catch(error => {
          this.$Message.error('获取数据失败！')
          console.log(error)
        })
      } else {
        this.getNodes(node.data.id).then(value => {
          this.$nextTick(() => {
            if (node.level === 1) {
              this.currentNode = node
            }
            this.nodeList = node.childNodes
          })
          this.tableData = value
          return resolve(value)
        })
      }
    },
    getNodes (parentId) {
      return new Promise((resolve, reject) => {
        const [url, httpConfig] = [
            `/api/getDmpAppCatalogsByParentId/${parentId}`
        ]
        this.$http.get(url, httpConfig).then(response => {
          resolve(response.data.data)
        }).catch(error => {
          this.$Message.error('获取数据失败！')
          reject(error)
        })
      })
    },
    clickNode (data, node, nodeCom) {
      this.currentNode = node
      this.nodeList = node.childNodes
      this.tableData = this.nodeList.map(item => {
        return {
          id: item.data.id,
          catalogname: item.data.catalogname
        }
      })
      // this.getNodes(data.id).then(value => {
      //   this.tableData = value
      // })
    },
    deleteRow (row, index) {
      this.showModal = true
      this.selectRow = row
      this.treeNodeIndex = index
    },
    confirmDel () {
      this.$Message.success('删除成功！')
      this.$refs.tree.remove(this.nodeList[this.treeNodeIndex])
      this.removeTableData(this.tableData, this.selectRow.id)
      this.selectRow = {}
      this.treeNodeIndex = -1
    },
    cancelDel () {
      this.selectRow = {}
      this.treeNodeIndex = -1
    },
    removeTableData (data, id) {
      const index = data.findIndex((item) => {
        return item.id === id
      })
      data.splice(index, 1)
    },
    addApp () {
      this.confirmButtonName = '创建'
      this.title = '添加应用'
      this.addmodalshow = true
      this.formType = 'add'
    },
    ok () {
      if (this.formType === 'add') {
        const newData = { ...this.editformData }
        this.tableData.push(newData)
        this.append(newData, this.currentNode)
        this.cancel()
        this.$Message.success('添加成功')
      } else {
        this.updateTable(this.tableData, this.editformData.id)
        this.currentNode.loaded = false
        this.currentNode.expand()
        this.cancel()
        this.$Message.success('修改成功')
      }
    },
    cancel () {
      this.addmodalshow = false
      this.editformData = {
        id: Math.ceil(Math.random() * 10 + 10),
        catalogname: null
      }
    },
    append (data, node) {
      this.$refs.tree.append(data, node)
    },
    edit (row, index) {
      this.confirmButtonName = '确定'
      this.title = '编辑应用'
      this.formType = 'edit'
      this.editformData.id = row.id
      this.editformData.catalogname = row.catalogname
      this.editformData.ordernum = row.ordernum
      this.addmodalshow = true
    },
    updateTable (table, id) {
      const index = table.findIndex((item) => {
        return item.id === id
      })
      table[index] = this.editformData
      this.$refs.tree.remove(table[index])
      this.$refs.tree.append(this.editformData, this.currentNode)
    },
    pageChange (currentPage) {
      this.currentPage = currentPage
    }
  }
}
</script>

    具体功能和上面的一样

    <template>
  <div class="appTypeManage" style="display: flex;flex-direction: column;height: 100%;">
    <alert></alert>
    <top-introduction></top-introduction>
    <Modal
      title="删除数据"
      :mask-closable="false"
      v-model="showModal"
      @on-ok="confirmDel"
      @on-cancel="cancelDel"
    >
      <p style="color: red;">确认删除该条信息?</p>
    </Modal>
    <Modal
      v-model="addmodalshow"
      :title="title"
      :mask-closable="false">
      <Form :model="editformData" :label-width="80">
        <FormItem label="分类名称">
          <Input v-model="editformData.catalogname"></Input>
        </FormItem>
        <FormItem label="备注">
          <Input v-model="editformData.remarks" type="textarea"></Input>
        </FormItem>
        <FormItem label="排序字段">
          <Input v-model="editformData.ordernum" type="number"></Input>
        </FormItem>
      </Form>
      <div slot="footer">
        <bottom-button :confirmButtonName="confirmButtonName" :hasMargin="false" @clickCreate="ok" @clickCancle="cancel"></bottom-button>
      </div>
    </Modal>
    <div class="container">
      <Card class="card-tree">
        <el-tree
          @node-click="clickNode"
          ref="tree"
          :props="props"
          node-key="id"
          :expand-on-click-node="true"
          :default-expanded-keys="defaultExpandsId"
          lazy
          :load="loadNode">
          <span class="custom-tree-node" slot-scope="{ node, data }">
            {{data.catalogname}}
          </span>
        </el-tree>
      </Card>
      <Card class="table-content">
        <Row style="text-align: right" class="margin-bottom-10">
          <Button type="primary" class="blue-button" @click="addApp">添加应用</Button>
        </Row>
        <Table :columns="columns1" :data="tableData.slice((currentPage-1)*5,5*currentPage)">
          <template slot-scope="{ row, index }" slot="action">
            <Button class="blue-button" type="primary" size="small" style="margin-right: 5px" @click="edit(row, index)">编辑</Button>
            <Button class="red-button" type="primary" size="small" @click="deleteRow(row,index)">删除</Button>
          </template>
        </Table>
        <Page style="text-align: right;" class="margin-top-10" :total="tableData.length" :page-size="5" @on-change="pageChange"/>

<script>
export default {
  name: 'appTypeManage',
  data () {
    return {
      props: {
        label: 'catalogname',
        children: 'children'
      },
      defaultExpandsId: [],
      columns1: [
        {
          type: 'index',
          title: '序号',
          align: 'center'
        },
        {
          title: '分类名称',
          key: 'catalogname',
          align: 'center'
        },
        {
          title: '备注',
          key: 'remarks',
          align: 'center'
        },
        {
          title: '操作',
          slot: 'action',
          width: 150,
          align: 'center'
        }
      ],
      tableData: [],
      showModal: false,
      selectRow: {},
      nodeList: [],
      treeNodeIndex: -1,
      currentNode: {},
      addmodalshow: false,
      formType: 'add',
      editformData: {
        id: null,
        catalogcode: null,
        catalogname: null,
        parentid: null,
        statue: null,
        creator: null,
        creatertime: null,
        modifytime: null,
        modifytor: null,
        remarks: null,
        ordernum: null,
        flag: null
      },
      currentPage: 1,
      title: '添加应用',
      confirmButtonName: '创建'
    }
  },
  components: {
    'alert': () => import('../components/top-alert/alert'),
    'bottomButton': () => import('../components/bottom-button/bottomButton'),
    'topIntroduction': () => import('../components/top-introduction/topIntroduction')
  },
  methods: {
    loadNode (node, resolve) {
      if (node.level === 0) {
        let root = {}
        this.$http.get(`/api/getDmpAppCatalogsByParentId/0`).then(response => {
          root = response.data.data[0]
          this.defaultExpandsId.push(root.id)
          resolve([root])
        }).catch(error => {
          this.$Message.error('获取数据失败！')
        })
      } else {
        this.getNodes(node.data.id).then(value => {
          this.$nextTick(() => {
            if (node.level === 1) {
              this.currentNode = node
            }
            this.nodeList = node.childNodes
          })
          this.tableData = value
          return resolve(value)
        })
      }
    },
    getNodes (parentId) {
      return new Promise((resolve, reject) => {
        const msg = this.$Message.loading({
          content: '正在获取数据',
          duration: 0
        })
        const [url, httpConfig] = [
          `/api/getDmpAppCatalogsByParentId/${parentId}`
        ]
        this.$http.get(url, httpConfig).then(response => {
          msg()
          resolve(response.data.data)
        }).catch(error => {
          msg()
          this.$Message.error('获取数据失败！')
          reject(error)
        })
      })
    },
    clickNode (data, node) {
      this.currentNode = node
      this.nodeList = node.childNodes
      this.getNodes(data.id).then(value => {
        this.tableData = value
      })
    },
    deleteRow (row, index) {
      this.showModal = true
      this.selectRow = row
      this.treeNodeIndex = index
    },
    confirmDel () {
      this.$http.delete(`/api/dmpAppCatalogs/${this.selectRow.id}`).then(resposne => {
        if (resposne.data.success) {
          this.$Message.success('删除成功！')
          this.$refs['tree'].remove(this.nodeList[this.treeNodeIndex])
          this.removeTableData(this.tableData, this.selectRow.id)
          this.selectRow = {}
          this.treeNodeIndex = -1
        }
      }).catch(error => {
        if (error.response) {
          this.$Message.error('删除失败！')
          this.selectRow = {}
          this.treeNodeIndex = -1
        }
      })
    },
    cancelDel () {
      this.selectRow = {}
      this.treeNodeIndex = -1
    },
    removeTableData (data, id) {
      let index = data.findIndex((item) => {
        return item.id === id
      })
      data.splice(index, 1)
    },
    addApp () {
      this.confirmButtonName = '创建'
      this.title = '添加应用'
      this.addmodalshow = true
      this.formType = 'add'
    },
    ok () {
      if (this.formType === 'add') {
        this.editformData.parentid = this.currentNode.key
        this.$http.post('/api/dmpAppCatalogs', this.editformData).then(response => {
          if (response.data.success) {
            this.tableData.push(response.data.data)
            this.append(response.data.data, this.currentNode)
            this.cancel()
            this.$Message.success(response.data.msg)
          } else {
            this.$Message.error(response.data.msg)
          }
        }).catch()
      } else {
        this.$http.put('/api/dmpAppCatalogs', this.editformData).then(response => {
          if (response.data.success) {
            this.currentNode.loaded = false
            this.currentNode.expand()
            this.cancel()
            this.$Message.success(response.data.msg)
          } else {
            this.$Message.error(response.data.msg)
          }
        }).catch()
      }
    },
    cancel () {
      this.addmodalshow = false
      this.editformData = {
        id: null,
        catalogcode: null,
        catalogname: null,
        parentid: null,
        statue: null,
        creator: null,
        creatertime: null,
        modifytime: null,
        modifytor: null,
        remarks: null,
        ordernum: null,
        flag: null
      }
    },
    append (data, node) {
      this.$refs['tree'].append(data, node)
    },
    edit (row, index) {
      this.confirmButtonName = '确定'
      this.title = '编辑应用'
      this.formType = 'edit'
      this.editformData.id = row.id
      this.editformData.catalogcode = row.catalogcode
      this.editformData.catalogname = row.catalogname
      this.editformData.parentid = row.parentid
      this.editformData.statue = row.statue
      this.editformData.creator = row.creator
      this.editformData.creatertime = row.creatertime
      this.editformData.modifytime = row.modifytime
      this.editformData.modifytor = row.modifytor
      this.editformData.remarks = row.remarks
      this.editformData.ordernum = row.ordernum
      this.editformData.flag = row.flag
      // this.addmodalshow2 = true
      this.addmodalshow = true
    },
    updateTable (table, id) {
      let index = table.findIndex((item) => {
        return item.id === id
      })
      table[index] = this.editformData
    },
    pageChange (currentPage) {
      this.currentPage = currentPage
    }
  }
}
</script>
