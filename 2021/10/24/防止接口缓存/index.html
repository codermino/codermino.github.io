    1. axios
      如果你们请求接口用的是axios的话，可以用路由拦截器来统一加，
      在config.url后面统一加时间戳就可以
      我们在请求数据的时候，会出现几种情况，判断权限或者loading的显隐。
      这些在每一个请求都会做的事情，我们可以封装一个service的请求函数，
      将axios+interceptors封装在内！
      (1)当有不符合的请求，拦截后取消请求！
      (2)当请求反映没有权限，重新去登录！
      这样就实现了所有请求的数据拦截和请求的响应拦截

    2. fetch
      这个项目里面封装了fetch请求的post,get方法，
      我只要在请求之前，把时间戳加进去就好了，
      get是加在url里面的，post是加在body里面

      // get post 两种方式的fetch
export const fetchPromise = (method, url, data, contentType) => {
  let headers = new Headers()
  let body = ''
  let config = {}
  if (method.toLowerCase() === 'get') {
    headers.append('Content-Type', 'text/plain')
    // get请求创建时间戳
    let time = new Date().getTime()
    url += '?time=' + time + '&'
    if (data) {
      // url += '?'
      for (let attr in data) {
        if (typeof data[attr] === 'string') {
          data[attr] = data[attr].replace(/%/g, '%25')
          data[attr] = data[attr].replace(/&/g, '%26')
          data[attr] = data[attr].replace(/\+/g, '%2B')
        }
        url += attr + '=' + data[attr] + '&'
      }
      url = url.substring(0, url.length - 1)
    }
    config = {
      method: 'get',
      headers: headers,
      credentials: 'same-origin'
    }
  } else {
    headers.append('Content-Type', 'application/x-www-form-urlencoded')
    if (data) {
      body = getQuery(data)
      // 给post请求加时间戳
      let time = new Date().getTime()
      body += 'time=' + time
    }
    config = {
      method: method,
      headers: headers,
      credentials: 'same-origin',
      body: body
    }
  }
  url = fullUrlAhead + url
  return new Promise((resolve, reject) => {
    fetch(url, config)
      .then((response) => {
        if (response.ok) {
          return response.json()
        } else {
          reject(response)
          return response
        }
      })
      .then((res) => {
        resolve(res)
      })
      .catch((err) => {
        reject(err)
      })
  })
}
