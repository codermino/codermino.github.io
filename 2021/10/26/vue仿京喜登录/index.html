    仿京喜的登录界面

    点击下面那个图标，
    之前点击的图标就替换对应位置的图标

    代码实现

    组件内部的代码不进行实现，只实现外面的大框

    <template>
    <div class="jingxi">
      <div class="content">
        <div class="content-top">
          <template v-for="item in bottomIconsList">
            <component :key="item.type" v-if="item.type === currentLoginType" :is="item.component"></component>
          </template>
        </div>
        <div class="content-bottom">
          <div class="content-bottom-other">
            <span>其他方式登录</span>
          </div>
          <div class="content-bottom-icon">
            <template v-for="(item, index) in bottomIconsList" >
              <i style="cursor: pointer;" @click="iconClick(item, index)" :key="item.type" v-show="currentLoginType !== item.type" :class="['iconfont', item.icon]"></i>
            </template>
          </div>
        </div>
      </div>
    </div>
</template>

<script>
import wxLogin from './jingxiCom/wxLogin'
import shoujiLogin from './jingxiCom/shoujiLogin'
import mimaLogin from './jingxiCom/mimaLogin'
import qqLogin from './jingxiCom/qqLogin'
import jdLogin from './jingxiCom/jdLogin'
export default {
  name: 'jingxi',
  components: {
    wxLogin,
    shoujiLogin,
    mimaLogin,
    qqLogin,
    jdLogin
  },
  data () {
    return {
      lastLoginType: 'jd',
      currentLoginType: 'jd',
      bottomIconsList: [
        {
          type: 'wx',
          icon: 'icon-weixin',
          component: 'wxLogin'
        },
        {
          type: 'sj',
          icon: 'icon-shouji1',
          component: 'shoujiLogin'
        },
        {
          type: 'zh',
          icon: 'icon-ATG-mimasz',
          component: 'mimaLogin'
        },
        {
          type: 'qq',
          icon: 'icon-11qq',
          component: 'qqLogin'
        },
        {
          type: 'jd',
          icon: 'icon-jingdong-',
          component: 'jdLogin'
        }
      ]
    }
  },
  methods: {
    iconClick (item, index) {
      if (this.currentLoginType !== this.lastLoginType) {
        this.lastLoginType = this.currentLoginType
      }
      const newBottomIconList = [...this.bottomIconsList]
      const clickItem = newBottomIconList[index]
      const lastItem = newBottomIconList.find(item => {
        return item.type === this.lastLoginType
      })
      if (!lastItem) {
        return
      }
      newBottomIconList[index] = lastItem
      newBottomIconList[newBottomIconList.length - 1] = clickItem
      this.bottomIconsList = newBottomIconList
      this.currentLoginType = item.type
    }
  }
}
</script>
