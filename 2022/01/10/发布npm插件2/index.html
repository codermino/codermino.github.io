    创建一个新项目，并且写好组件

    项目目录如下图所示:

    在组件文件夹下添加index.js

    导出和注册对应组件

    import vueCascade from './index.vue';
//.name就是开始说的vue文件暴露出来的name名，ldcPagination整个组件
vueCascade.install = Vue => Vue.component(vueCascade.name, vueCascade);
export default vueCascade;

    在组件父文件夹下添加index.js

    批量注册组件，同时也是打包的入口文件

    import vueCascade from './vue-cascade/index.js'
const components = [
    vueCascade
]

const install = function(Vue, opts = {}) {
    components.forEach(component => {
        Vue.component(component.name, component);
    });
}

if (typeof window !== 'undefined' && window.Vue) {
    install(window.Vue);
}

export default {
    install,
    vueCascade
}
